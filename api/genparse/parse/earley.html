154        if len(x) == 0:

163    def next_column(self, prev_cols, token):
164        prev_col = prev_cols[-1]
165        next_col = Column(prev_cols[-1].k + 1)
166        next_col_c_chart = next_col.c_chart
167        prev_col_i_chart = prev_col.i_chart
168
169        rest_Ys = self.rest_Ys
170        _update = self._update

172        Q = LocatorMaxHeap()
173
174        # SCAN: phrase(I, X/Ys, K) += phrase(I, X/[Y|Ys], J) * word(J, Y, K)
175        for item in prev_col.waiting_for[token]:
176            (I, X, Ys) = item
177            _update(next_col, Q, I, X, rest_Ys[Ys], prev_col_i_chart[item])

179        # ATTACH: phrase(I, X/Ys, K) += phrase(I, X/[Y|Ys], J) * phrase(J, Y/[], K)

188                (I, X, Ys) = customer
189                _update(next_col, Q, I, X, rest_Ys[Ys], col_J_i_chart[customer] * y)

195    def PREDICT(self, col):
196        # PREDICT: phrase(K, X/Ys, K) += rule(X -> Ys) with some filtering heuristics
197        k = col.k
198
199        # Filtering heuristic: Don't create the predicted item (K, X, [...], K)
200        # unless there exists an item that wants the X item that it may
201        # eventually provide.  In other words, for predicting this item to be
202        # useful there must be an item of the form (I, X', [X, ...], K) in this
203        # column for which lc(X', X) is true.
204        if col.k == 0:
205            agenda = [self.cfg.S]
206        else:
207            agenda = list(col.waiting_for)
208
209        outgoing = self.R_outgoing
210
211        reachable = set(agenda)
212
213        while agenda:
214            X = agenda.pop()
215            for Y in outgoing[X]:
216                if Y not in reachable:
217                    reachable.add(Y)
218                    agenda.append(Y)
219
220        rhs = self.rhs
221        _update = self._update
222        for X in reachable:
223            for w, Ys in rhs.get(X, ()):
224                _update(col, None, k, X, Ys, w)

226    def _update(self, col, Q, I, X, Ys, value):
227        K = col.k
228        if Ys == 0:
229            # Items of the form phrase(I, X/[], K)
230            item = (I, X)
231            was = col.c_chart.get(item)
232            if was is None:
233                Q[item] = -((K - I) * self.ORDER_MAX + self.order[X])
234                col.c_chart[item] = value
235            else:
236                col.c_chart[item] = was + value
237
238        else:
239            # Items of the form phrase(I, X/[Y|Ys], K)
240            item = (I, X, Ys)
241            was = col.i_chart.get(item)
242            if was is None:
243                col.waiting_for[self.first_Ys[Ys]].append(item)
244                col.i_chart[item] = value
245            else:
246                col.i_chart[item] = was + value

248    # We have derived the `next_token_weights` algorithm by backpropagation on
249    # the program with respect to the item `phrase(0, s, K)`.
250    #
251    # ATTACH: phrase(I, X/Ys, K) += phrase(I, X/[Y|Ys], J) * phrase(J, Y/[], K)
252    #
253    # Directly applying the gradient transformation, we get
254    #
255    # âˆ‡phrase(0, s/[], K) += 1
256    # âˆ‡phrase(J, Y/[], K) += phrase(I, X/[Y|Ys], J) * âˆ‡phrase(I, X/Ys, K)
257    #
258    # Some quick analysis reveals that the `Ys` list must always be [], and
259    # that K is always equal to the final column.  We specialize the program
260    # below:
261    #
262    # âˆ‡phrase(0, s/[], K) += 1
263    # âˆ‡phrase(J, Y/[], K) += phrase(I, X/[Y], J) * âˆ‡phrase(I, X/[], K)
264    #
265    # We can abbreviate the names:
266    #
267    # q(0, s) += 1
268    # q(J, Y) += phrase(I, X/[Y], J) * q(I, X)
269    #
270    # These items satisfy (I > J) and (X > Y) where the latter is the
271    # nonterminal ordering.  Thus, we can efficiently evaluate these equations
272    # by backward chaining.
273    #
274    # The final output is the vector
275    #
276    # p(W) += q(I, X) * phrase(I, X/[W], J)  where len(J) * terminal(W).
277    #
278    def next_token_weights(self, cols):
279        is_terminal = self.cfg.is_terminal
280        zero = self.cfg.R.zero
281
282        q = {}
283        q[0, self.cfg.S] = self.cfg.R.one
284
285        col = cols[-1]
286        col_waiting_for = col.waiting_for
287        col_i_chart = col.i_chart
288
289        # SCAN: phrase(I, X/Ys, K) += phrase(I, X/[Y|Ys], J) * word(J, Y, K)
290        p = self.cfg.R.chart()
291
292        for Y in col_waiting_for:
293            if is_terminal(Y):
294                total = zero
295                for I, X, Ys in col_waiting_for[Y]:
296                    if self.unit_Ys[Ys]:
297                        node = (I, X)
298                        value = self._helper(node, cols, q)
299                        total += col_i_chart[I, X, Ys] * value
300                p[Y] = total
301
302        return p

304    def _helper(self, top, cols, q):
305        value = q.get(top)
306        if value is not None:
307            return value
308
309        zero = self.cfg.R.zero
310        stack = [Node(top, None, zero)]
311
312        while stack:
313            node = stack[-1]  # ðŸ‘€
314
315            # place neighbors above the node on the stack
316            (J, Y) = node.node
317
318            t = node.cursor
319
320            if node.edges is None:
321                node.edges = [x for x in cols[J].waiting_for[Y] if self.unit_Ys[x[2]]]
322
323            # cursor is at the end, all neighbors are done
324            elif t == len(node.edges):
325                # clear the node from the stack
326                stack.pop()
327                # promote the incomplete value node.value to a complete value (q)
328                q[node.node] = node.value
329
330            else:
331                (I, X, _) = arc = node.edges[t]
332                neighbor = (I, X)
333                neighbor_value = q.get(neighbor)
334                if neighbor_value is None:
335                    stack.append(Node(neighbor, None, zero))
336                else:
337                    # neighbor value is ready, advance the cursor, add the
338                    # neighbors contribution to the nodes value
339                    node.cursor += 1
340                    node.value += cols[J].i_chart[arc] * neighbor_value
341
342        return q[top]

We say that $p\colon V^* \to [0,1]$ is a language model if $p$ is a probability distribution over strings from some alphabet $V$ of tokens.

Every language model admits a left-to-right factorization:

$$ p(x_1 x_2 \cdots x_T) = p(x_1 \mid \varepsilon) p(x_2 \mid x_1) \cdots p(x_T \mid x_1 \cdots x_{T-1}) p(\mathrm{EOS} \mid x_1 \cdots x_T) $$

Arguments:
  • - V: a vocabulary of symbols
  • - eos: a distinguished end of sequence symbol
  • - p_next(xs): $p(\cdot \mid x_1 \cdots x_T)$ is provided by subclasses.

196    def PREDICT(self, col):
197        # PREDICT: phrase(K, X/Ys, K) += rule(X -> Ys) with some filtering heuristics
198        k = col.k
199
200        # Filtering heuristic: Don't create the predicted item (K, X, [...], K)
201        # unless there exists an item that wants the X item that it may
202        # eventually provide.  In other words, for predicting this item to be
203        # useful there must be an item of the form (I, X', [X, ...], K) in this
204        # column for which lc(X', X) is true.
205        if col.k == 0:
206            agenda = [self.cfg.S]
207        else:
208            agenda = list(col.waiting_for)
209
210        outgoing = self.R_outgoing
211
212        reachable = set(agenda)
213
214        while agenda:
215            X = agenda.pop()
216            for Y in outgoing[X]:
217                if Y not in reachable:
218                    reachable.add(Y)
219                    agenda.append(Y)
220
221        rhs = self.rhs
222        _update = self._update
223        for X in reachable:
224            for w, Ys in rhs.get(X, ()):
225                _update(col, None, k, X, Ys, w)

227    def _update(self, col, Q, I, X, Ys, value):
228        K = col.k
229        if Ys == 0:
230            # Items of the form phrase(I, X/[], K)
231            item = (I, X)
232            was = col.c_chart.get(item)
233            if was is None:
234                Q[item] = -((K - I) * self.ORDER_MAX + self.order[X])
235                col.c_chart[item] = value
236            else:
237                col.c_chart[item] = was + value
238
239        else:
240            # Items of the form phrase(I, X/[Y|Ys], K)
241            item = (I, X, Ys)
242            was = col.i_chart.get(item)
243            if was is None:
244                col.waiting_for[self.first_Ys[Ys]].append(item)
245                col.i_chart[item] = value
246            else:
247                col.i_chart[item] = was + value

249    # We have derived the `next_token_weights` algorithm by backpropagation on
250    # the program with respect to the item `phrase(0, s, K)`.
251    #
252    # ATTACH: phrase(I, X/Ys, K) += phrase(I, X/[Y|Ys], J) * phrase(J, Y/[], K)
253    #
254    # Directly applying the gradient transformation, we get
255    #
256    # âˆ‡phrase(0, s/[], K) += 1
257    # âˆ‡phrase(J, Y/[], K) += phrase(I, X/[Y|Ys], J) * âˆ‡phrase(I, X/Ys, K)
258    #
259    # Some quick analysis reveals that the `Ys` list must always be [], and
260    # that K is always equal to the final column.  We specialize the program
261    # below:
262    #
263    # âˆ‡phrase(0, s/[], K) += 1
264    # âˆ‡phrase(J, Y/[], K) += phrase(I, X/[Y], J) * âˆ‡phrase(I, X/[], K)
265    #
266    # We can abbreviate the names:
267    #
268    # q(0, s) += 1
269    # q(J, Y) += phrase(I, X/[Y], J) * q(I, X)
270    #
271    # These items satisfy (I > J) and (X > Y) where the latter is the
272    # nonterminal ordering.  Thus, we can efficiently evaluate these equations
273    # by backward chaining.
274    #
275    # The final output is the vector
276    #
277    # p(W) += q(I, X) * phrase(I, X/[W], J)  where len(J) * terminal(W).
278    #
279    def next_token_weights(self, cols):
280        is_terminal = self.cfg.is_terminal
281        zero = self.cfg.R.zero
282
283        q = {}
284        q[0, self.cfg.S] = self.cfg.R.one
285
286        col = cols[-1]
287        col_waiting_for = col.waiting_for
288        col_i_chart = col.i_chart
289
290        # SCAN: phrase(I, X/Ys, K) += phrase(I, X/[Y|Ys], J) * word(J, Y, K)
291        p = self.cfg.R.chart()
292
293        for Y in col_waiting_for:
294            if is_terminal(Y):
295                total = zero
296                for I, X, Ys in col_waiting_for[Y]:
297                    if self.unit_Ys[Ys]:
298                        node = (I, X)
299                        value = self._helper(node, cols, q)
300                        total += col_i_chart[I, X, Ys] * value
301                p[Y] = total
302
303        return p

305    def _helper(self, top, cols, q):
306        value = q.get(top)
307        if value is not None:
308            return value
309
310        zero = self.cfg.R.zero
311        stack = [Node(top, None, zero)]
312
313        while stack:
314            node = stack[-1]  # ðŸ‘€
315
316            # place neighbors above the node on the stack
317            (J, Y) = node.node
318
319            t = node.cursor
320
321            if node.edges is None:
322                node.edges = [x for x in cols[J].waiting_for[Y] if self.unit_Ys[x[2]]]
323
324            # cursor is at the end, all neighbors are done
325            elif t == len(node.edges):
326                # clear the node from the stack
327                stack.pop()
328                # promote the incomplete value node.value to a complete value (q)
329                q[node.node] = node.value
330
331            else:
332                (I, X, _) = arc = node.edges[t]
333                neighbor = (I, X)
334                neighbor_value = q.get(neighbor)
335                if neighbor_value is None:
336                    stack.append(Node(neighbor, None, zero))
337                else:
338                    # neighbor value is ready, advance the cursor, add the
339                    # neighbors contribution to the nodes value
340                    node.cursor += 1
341                    node.value += cols[J].i_chart[arc] * neighbor_value
342
343        return q[top]
