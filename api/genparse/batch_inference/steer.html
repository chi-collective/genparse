 12class BatchStepModel:
 13    """
 14    A model for performing batch steps during inference.
 15
 16    Attributes:
 17        batch_proposal (BatchProposal): The batch proposal.
 18        batch_llm (BatchLLM): The batch language model.
 19        max_tokens (int): The maximum number of tokens to sample.
 20
 21    """
 22
 23    def __init__(self, batch_proposal, batch_llm, max_tokens, prompt=None):
 24        self.batch_proposal = batch_proposal
 25        self.batch_llm = batch_llm
 26        self.max_tokens = max_tokens
 27
 28        if prompt is not None:
 29            self.set_prompt(prompt)
 30
 31        atexit.register(self.cleanup)
 32
 33    def set_prompt(self, prompt):
 34        """
 35        Set the prompt for the batch language model.
 36
 37        Args:
 38            prompt (str): The prompt to set.
 39        """
 40        self.batch_llm.set_prompt(prompt)
 41
 42    def batch_step(self, particles, is_initial=False):
 43        """
 44        Perform a batch step during inference.
 45        Computes the next token logprobs for each particle.
 46        Samples an extension for each particle and updates the particle weights.
 47
 48        Args:
 49            particles (list): List of Particle objects.
 50            is_initial (bool, optional): Flag indicating if it is the initial step. Defaults to False.
 51
 52        Returns:
 53            list: List of updated Particle objects after the batch step.
 54        """
 55        logprobs, particle_idx_to_logprob_idx = self.batch_llm.batch_next_token_logprobs(
 56            particles=particles, is_initial=is_initial
 57        )
 58
 59        extensions, extension_id_to_particle_id = (
 60            self.batch_proposal.batch_particle_extensions(
 61                particles=particles,
 62                logprobs=logprobs,
 63                particle_idx_to_logprob_idx=particle_idx_to_logprob_idx,
 64            )
 65        )
 66
 67        for extension_id, particle_id in extension_id_to_particle_id.items():
 68            particle = particles[particle_id]
 69            extension = extensions[extension_id]
 70            particles[particle_id] = Particle(
 71                prompt=particle.prompt,
 72                log_weight=particle.log_weight + extension.log_weight,
 73                log_potential=particle.log_potential,
 74                context=particle.context + (extension.token,),
 75                context_ids=particle.context_ids + (extension.token_id,),
 76                done=(
 77                    extension.token == self.batch_proposal.eos
 78                    or extension.token_id == self.batch_llm.eos_token_id
 79                    or len(particle.context) + 1 >= self.max_tokens
 80                ),
 81                parent=particle.parent,
 82            )
 83
 84        return particles
 85
 86    def cleanup(self):
 87        """
 88        Clean up resources used by the batch proposal and batch language model.
 89        """
 90        self.batch_proposal.cleanup()
 91        self.batch_llm.cleanup()

118    def twist(self, log_potential):
119        if self.log_potential == -np.inf:
120            assert log_potential == -np.inf, (
121                'Potentials  must satisfy (x) = 0 => (xy) = 0, forall x,y in V*'
122            )
123            self.log_weight = -np.inf
124        else:
125            self.log_weight += log_potential - self.log_potential
126            self.log_potential = log_potential
127
128    def untwist(self):
129        self.log_weight -= self.log_potential
130        self.log_potential = 0
131
132    def _replace(self, **kwargs):
133        for k, v in kwargs.items():
134            setattr(self, k, v)

146class ParticleApproximation:
147    """
148    A particle approximation for a target distribution.
149
150    Attributes:
151        particles (list): List of particles representing the distribution.
152        size (int): Number of particles in the approximation.
153        log_weights (ndarray): Array of log weights for each particle.
154        log_total (float): Log of the sum of the weights.
155        log_ml (float): Log marginal likelihood estimate.
156        log_normalized_weights (ndarray): Array of log-normalized weights.
157        log_ess (float): Log of the effective sample size.
158        ess (float): Effective sample size, estimated as 1 / sum of squared normalized weights.
159        record (SMCRecord or None): Record associated with the approximation, if provided.
160    """
161
162    def __init__(self, particles, record=None):
163        self.particles = list(particles)
164        self.size = len(particles)
165        self.log_weights = np.array([p.log_weight for p in self.particles])
166        self.log_total = logsumexp(self.log_weights)
167
168        # log-marginal likelihood estimate (Note: need to exponentiate to have
169        # an unbiased estimate of the true marginal likelihood).
170        self.log_ml = np.log(np.mean(np.exp(self.log_weights)))
171
172        # log-normalized weights
173        self.log_normalized_weights = self.log_weights - self.log_total
174
175        # Compute the effective sample size
176        self.log_ess = -logsumexp(2 * self.log_normalized_weights)
177        self.ess = np.exp(self.log_ess)
178
179        self.record = record
180
181    @property
182    def posterior(self):
183        posterior = Float.chart()
184        for p, prob in zip(self.particles, np.exp(self.log_normalized_weights)):
185            posterior[''.join(p.context)] += prob
186        return posterior.normalize().sort_descending()
187
188    def __len__(self):
189        return self.size
190
191    def __iter__(self):
192        return iter(self.particles)
193
194    def __getitem__(self, i):
195        return self.particles[i]
196
197    def sample(self, n=None, draw=sample_dict):
198        if n is None:
199            return draw(self.posterior)
200        else:
201            return [draw(self.posterior) for _ in range(n)]
202
203    def __str__(self):
204        return str(self.posterior)
205
206    def _repr_html_(self):
207        return self.posterior._repr_html_()
208
209    def risk(self, kernel, candidate):
210        return sum(p * kernel(candidate, y) for y, p in self.posterior.items())
211
212    def finalize(self, eos):
213        "Optionally, we can zero-out invalid particles, i.e., those that do not end in eos."
214        return ParticleApproximation(
215            [
216                p if p.context[-1] == eos else p._replace(log_weight=float('-inf'))
217                for p in self
218            ],
219            self.record,
220        )
221
222    def show(self):
223        for p in sorted(self, reverse=True):
224            print(p)

239def stratified_resample(weights):
240    # source: https://filterpy.readthedocs.io/en/latest/_modules/filterpy/monte_carlo/resampling.html#stratified_resample
241    """Performs the stratified resampling algorithm used by particle filters.
242
243    This algorithms aims to make selections relatively uniformly across the
244    particles. It divides the cumulative sum of the weights into N equal
245    divisions, and then selects one particle randomly from each division. This
246    guarantees that each sample is between 0 and 2/N apart.
247
248    Parameters
249    ----------
250    weights : list-like of float
251        list of weights as floats
252
253    Returns
254    -------
255
256    indexes : ndarray of ints
257        array of indexes into the weights defining the resample. i.e. the
258        index of the zeroth resample is indexes[0], etc.
259    """
260
261    N = len(weights)
262    # make N subdivisions, and chose a random position within each one
263    positions = (random(N) + range(N)) / N
264
265    indexes = np.zeros(N, 'i')
266    cumulative_sum = np.cumsum(weights)
267    i, j = 0, 0
268    while i < N:
269        if positions[i] < cumulative_sum[j]:
270            indexes[i] = j
271            i += 1
272        else:
273            j += 1
274    return indexes

277def maybe_resample(
278    particles,
279    ess_threshold,
280    return_record,
281    step_info,
282    verbosity,
283    resample_method='multinomial',
284):
285    """
286    Resamples particles if the effective sample size (ESS) falls below a threshold.
287
288    If a resampling step occurs, all particle weights are set to the average weight of the particles prior to resampling.
289
290    Args:
291        particles (list): List of particles.
292        ess_threshold (float): Threshold for the effective sample size, specified as a fraction of the number of particles.
293        resample_method (str): Resampling method to use. Either 'multinomial' or 'stratified'. Default is 'multinomial'.
294        return_record (bool): Flag indicating whether to log additional information in the step_info dictionary.
295        step_info (dict): Dictionary to log step information.
296        verbosity (int): Verbosity level.
297
298    Returns:
299        list: List of resampled particles.
300
301    """
302    if return_record:
303        step_info['particles'] = [
304            {
305                'context': p.context,
306                'weight': p.log_weight,
307                'context_ids': p.context_ids,
308            }
309            for p in particles
310        ]
311
312    n_particles = len(particles)
313    log_weights = [p.log_weight for p in particles]
314    log_total = logsumexp(log_weights)
315    avg_log_weight = log_total - np.log(n_particles)
316
317    if return_record or verbosity > 0:
318        step_info['average_weight'] = avg_log_weight
319
320    log_normalized_weights = log_weights - log_total
321    log_ess = -logsumexp(2 * log_normalized_weights)
322    ess = np.exp(log_ess)
323
324    if verbosity > 0:
325        pretty_print_particles(particles, step_info)
326
327    if ess < n_particles * ess_threshold:
328        if resample_method == 'multinomial':
329            indices = log_sample(log_normalized_weights, size=n_particles)
330        elif resample_method == 'stratified':
331            indices = stratified_resample(np.exp(log_normalized_weights))
332        else:
333            raise ValueError(f'Unknown resampling method: {resample_method}')
334
335        particles = [
336            copy.deepcopy(particles[i])._replace(log_weight=avg_log_weight, parent=i)
337            for i in indices
338        ]
339
340        if return_record:
341            step_info['resample_indices'] = indices.tolist()
342
343        if verbosity > 0:
344            print(
345                f'  Resampling! {indices}. Weights all set to = {avg_log_weight:.4f}.'
346            )
347    else:
348        if verbosity > 0:
349            print('')
350
351    return particles

354def twist_particles(particles, potential):
355    if potential is None:
356        raise ValueError('Potential function must be provided to twist particles.')
357    log_potentials = potential(particles)
358    for i, p in enumerate(particles):
359        p.twist(log_potentials[i])
360    return particles

363def smc(
364    batch_model,
365    n_particles,
366    potential=None,
367    ess_threshold=0.5,
368    verbosity=0,
369    return_record=False,
370    resample_method='multinomial',
371):
372    """Standard sequential Monte Carlo algorithm with multinomial resampling.
373
374    Args:
375      - `batch_model` (`BatchStepper`): The model to perform inference on.
376      - `n_particles` (`int`): The number of particles to perform inference with.
377      - `potential` (`Callable`):
378            A function that when called on a list of particles, returns a list with the log potential values
379            for each particle in the list. Optional.
380      - `ess_threshold` (`float`): Effective sample size below which resampling
381         triggered, given as a fraction of `n_particles`. Default is 0.5.
382      - `verbosity` (`int`): Verbosity level. When > 0, particles are printed at each step. Default is 0.
383      - `return_record` (`bool`): Flag indicating whether to return a record of the inference steps. Default is False.
384      - `resample_method` (`str`): Resampling method to use. Either 'multinomial' or 'stratified'. Default is 'multinomial'.
385
386    Returns:
387      - `particle_approximation` (`ParticleApproximation`): The completed particle approximation.
388    """
389
390    record = (
391        SMCRecord(
392            {
393                'n_particles': n_particles,
394                'ess_threshold': ess_threshold,
395                'resample_method': resample_method,
396                'algorithm': 'smc_standard_record',
397                'history': [],
398            }
399        )
400        if return_record
401        else None
402    )
403
404    step_num = 1
405    particles = init_particles(n_particles)
406    while not all(p.done for p in particles):
407        step_info = {'step': step_num}
408
409        particles = batch_model.batch_step(particles, is_initial=step_num == 1)
410
411        if (potential is not None) and (ess_threshold > 0):
412            particles = twist_particles(particles, potential)
413
414        particles = maybe_resample(
415            particles=particles,
416            ess_threshold=ess_threshold,
417            return_record=return_record,
418            step_info=step_info,
419            verbosity=verbosity,
420            resample_method=resample_method,
421        )
422
423        if return_record:
424            record['history'].append(step_info)
425
426        step_num += 1
427
428    if (potential is not None) and (ess_threshold == 0):
429        particles = twist_particles(particles, potential)
430
431    return ParticleApproximation(particles, record)

240def stratified_resample(weights):
241    # source: https://filterpy.readthedocs.io/en/latest/_modules/filterpy/monte_carlo/resampling.html#stratified_resample
242    """Performs the stratified resampling algorithm used by particle filters.
243
244    This algorithms aims to make selections relatively uniformly across the
245    particles. It divides the cumulative sum of the weights into N equal
246    divisions, and then selects one particle randomly from each division. This
247    guarantees that each sample is between 0 and 2/N apart.
248
249    Parameters
250    ----------
251    weights : list-like of float
252        list of weights as floats
253
254    Returns
255    -------
256
257    indexes : ndarray of ints
258        array of indexes into the weights defining the resample. i.e. the
259        index of the zeroth resample is indexes[0], etc.
260    """
261
262    N = len(weights)
263    # make N subdivisions, and chose a random position within each one
264    positions = (random(N) + range(N)) / N
265
266    indexes = np.zeros(N, 'i')
267    cumulative_sum = np.cumsum(weights)
268    i, j = 0, 0
269    while i < N:
270        if positions[i] < cumulative_sum[j]:
271            indexes[i] = j
272            i += 1
273        else:
274            j += 1
275    return indexes
