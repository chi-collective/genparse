169    def epsremove(self):
170        "Return an equivalent machine with no epsilon arcs."
171        E = self.E
172        S = E.closure()
173        new = self.spawn(keep_stop=True)
174        for i, w_i in self.I:
175            for k in S.outgoing[i]:
176                new.add_I(k, w_i * S[i, k])
177        for i, a, j, w_ij in self.arcs():
178            if a == EPSILON:
179                continue
180            for k in S.outgoing[j]:
181                new.add_arc(i, a, k, w_ij * S[j, k])
182        return new

358    @cached_property
359    def push(self):
360        "Weight pushing algorithm (Mohri, 2001); assumes v**-1 possible for weights."
361        V = self.backward
362        new = self.spawn()
363        for i in self.states:
364            if V[i] == self.R.zero:
365                continue
366            new.add_I(i, self.start[i] * V[i])
367            new.add_F(i, V[i] ** (-1) * self.stop[i])
368            for a, j, w in self.arcs(i):
369                new.add_arc(i, a, j, V[i] ** (-1) * w * V[j])
370        return new

374        """

377        However, it is generally slower to evaluation.

427    @cached_property
428    def determinize(self):
429        """
430        Mohri (2009)'s "on-the-fly" determinization semi-algorithm.
431        https://link.springer.com/chapter/10.1007/978-3-642-01492-5_6
432
433        Use with caution as this method may not terminate.
434        """
435
436        self = self.epsremove.push
437
438        def _powerarcs(Q):
439            U = {a: self.R.chart() for a in self.alphabet}
440
441            for i, u in Q.items():
442                for a, j, v in self.arcs(i):
443                    U[a][j] += u * v
444
445            for a in U:
446                R = U[a]
447                W = sum(R.values(), start=self.R.zero)
448
449                if 0:
450                    # If we cannot extract a common factor, then all of the arcs will have weight one
451                    yield a, frozendict(R), self.R.one
452
453                else:
454                    yield a, frozendict({p: W ** (-1) * R[p] for p in R}), W
455
456        D = self.spawn()
457
458        stack = []
459        visited = set()
460
461        Q = frozendict({i: w for i, w in self.I})
462        D.add_I(Q, self.R.one)
463        stack.append(Q)
464        visited.add(Q)
465
466        while stack:
467            P = stack.pop()
468            for a, Q, w in _powerarcs(P):
469                if Q not in visited:
470                    stack.append(Q)
471                    visited.add(Q)
472                D.add_arc(P, a, Q, w)
473
474        for Q in D.states:
475            for q in Q:
476                D.add_F(Q, Q[q] * self.stop[q])
477
478        return D

480    def to_cfg(self, S=None, recursion='right'):
481        """
482        Convert the WFSA to a WCFG with the same weighted language.
483
484        The option `recursion` in {"left", "right"} specifies whether the WCFG
485        should be left or right recursive.
486
487        """
488        from genparse.cfg import CFG, _gen_nt
489
490        if S is None:
491            S = _gen_nt()
492        cfg = CFG(R=self.R, V=self.alphabet - {EPSILON}, S=S)
493
494        if recursion == 'right':
495            # add production rule for initial states
496            for i, w in self.I:
497                cfg.add(w, S, i)
498
499            # add production rule for final states
500            for i, w in self.F:
501                cfg.add(w, i)
502
503            # add other production rules
504            for i, a, j, w in self.arcs():
505                if a == EPSILON:
506                    cfg.add(w, i, j)
507                else:
508                    cfg.add(w, i, a, j)
509
510        else:
511            assert recursion == 'left'
512
513            # add production rule for final states
514            for i, w in self.F:
515                cfg.add(w, S, i)
516
517            # add production rule for initial states
518            for i, w in self.I:
519                cfg.add(w, i)
520
521            # add other production rules
522            for i, a, j, w in self.arcs():
523                if a == EPSILON:
524                    cfg.add(w, j, i)
525                else:
526                    cfg.add(w, j, i, a)
527
528        return cfg

428    @cached_property
429    def determinize(self):
430        """
431        Mohri (2009)'s "on-the-fly" determinization semi-algorithm.
432        https://link.springer.com/chapter/10.1007/978-3-642-01492-5_6
433
434        Use with caution as this method may not terminate.
435        """
436
437        self = self.epsremove.push
438
439        def _powerarcs(Q):
440            U = {a: self.R.chart() for a in self.alphabet}
441
442            for i, u in Q.items():
443                for a, j, v in self.arcs(i):
444                    U[a][j] += u * v
445
446            for a in U:
447                R = U[a]
448                W = sum(R.values(), start=self.R.zero)
449
450                if 0:
451                    # If we cannot extract a common factor, then all of the arcs will have weight one
452                    yield a, frozendict(R), self.R.one
453
454                else:
455                    yield a, frozendict({p: W ** (-1) * R[p] for p in R}), W
456
457        D = self.spawn()
458
459        stack = []
460        visited = set()
461
462        Q = frozendict({i: w for i, w in self.I})
463        D.add_I(Q, self.R.one)
464        stack.append(Q)
465        visited.add(Q)
466
467        while stack:
468            P = stack.pop()
469            for a, Q, w in _powerarcs(P):
470                if Q not in visited:
471                    stack.append(Q)
472                    visited.add(Q)
473                D.add_arc(P, a, Q, w)
474
475        for Q in D.states:
476            for q in Q:
477                D.add_F(Q, Q[q] * self.stop[q])
478
479        return D

481    def to_cfg(self, S=None, recursion='right'):
482        """
483        Convert the WFSA to a WCFG with the same weighted language.
484
485        The option `recursion` in {"left", "right"} specifies whether the WCFG
486        should be left or right recursive.
487
488        """
489        from genparse.cfg import CFG, _gen_nt
490
491        if S is None:
492            S = _gen_nt()
493        cfg = CFG(R=self.R, V=self.alphabet - {EPSILON}, S=S)
494
495        if recursion == 'right':
496            # add production rule for initial states
497            for i, w in self.I:
498                cfg.add(w, S, i)
499
500            # add production rule for final states
501            for i, w in self.F:
502                cfg.add(w, i)
503
504            # add other production rules
505            for i, a, j, w in self.arcs():
506                if a == EPSILON:
507                    cfg.add(w, i, j)
508                else:
509                    cfg.add(w, i, a, j)
510
511        else:
512            assert recursion == 'left'
513
514            # add production rule for final states
515            for i, w in self.F:
516                cfg.add(w, S, i)
517
518            # add production rule for initial states
519            for i, w in self.I:
520                cfg.add(w, i)
521
522            # add other production rules
523            for i, a, j, w in self.arcs():
524                if a == EPSILON:
525                    cfg.add(w, j, i)
526                else:
527                    cfg.add(w, j, i, a)
528
529        return cfg

432        https://link.springer.com/chapter/10.1007/978-3-642-01492-5_6
433
434        Use with caution as this method may not terminate.
435        """
436
437        self = self.epsremove.push
438
439        def _powerarcs(Q):
440            U = {a: self.R.chart() for a in self.alphabet}
441
442            for i, u in Q.items():
443                for a, j, v in self.arcs(i):
444                    U[a][j] += u * v
445
446            for a in U:
447                R = U[a]
448                W = sum(R.values(), start=self.R.zero)
449
450                if 0:
451                    # If we cannot extract a common factor, then all of the arcs will have weight one
452                    yield a, frozendict(R), self.R.one
453
454                else:
455                    yield a, frozendict({p: W ** (-1) * R[p] for p in R}), W
456
457        D = self.spawn()
458
459        stack = []
460        visited = set()
461
462        Q = frozendict({i: w for i, w in self.I})
463        D.add_I(Q, self.R.one)
464        stack.append(Q)
465        visited.add(Q)
466
467        while stack:
468            P = stack.pop()
469            for a, Q, w in _powerarcs(P):
470                if Q not in visited:
471                    stack.append(Q)
472                    visited.add(Q)
473                D.add_arc(P, a, Q, w)
474
475        for Q in D.states:
476            for q in Q:
477                D.add_F(Q, Q[q] * self.stop[q])
478
