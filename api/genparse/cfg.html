 126    @classmethod
 127    def from_string(
 128        cls,
 129        string,
 130        semiring,
 131        comment='#',
 132        start='S',
 133        is_terminal=lambda x: not x[0].isupper(),
 134    ):
 135        V = set()
 136        cfg = cls(R=semiring, S=start, V=V)
 137        string = string.replace('->', '')  # synonym for the arrow
 138        for line in string.split('\n'):
 139            line = line.strip()
 140            if not line or line.startswith(comment):
 141                continue
 142            try:
 143                [(w, lhs, rhs)] = re.findall(r'(.*):\s*(\S+)\s*\s*(.*)$', line)
 144                lhs = lhs.strip()
 145                rhs = rhs.strip().split()
 146                for x in rhs:
 147                    if is_terminal(x):
 148                        V.add(x)
 149                cfg.add(semiring.from_string(w), lhs, *rhs)
 150            except ValueError:
 151                raise ValueError(f'bad input line:\n{line}')  # pylint: disable=W0707
 152        return cfg

 164    def __call__(self, xs):
 165        "Compute the total weight of the `xs` sequence."
 166        self = self.cnf  # need to do this here because the start symbol might change
 167        return self._parse_chart(xs)[0, self.S, len(xs)]

 169    def _parse_chart(self, xs):
 170        "Implements CKY algorithm for evaluating the total weight of the `xs` sequence."
 171        (nullary, terminal, binary) = self._cnf  # will convert to CNF
 172        N = len(xs)
 173        # nullary rule
 174        c = self.R.chart()
 175        for i in range(N + 1):
 176            c[i, self.S, i] += nullary
 177        # preterminal rules
 178        for i in range(N):
 179            for r in terminal[xs[i]]:
 180                c[i, r.head, i + 1] += r.w
 181        # binary rules
 182        for span in range(2, N + 1):
 183            for i in range(N - span + 1):
 184                k = i + span
 185                for j in range(i + 1, k):
 186                    for r in binary:
 187                        X, [Y, Z] = r.head, r.body
 188                        c[i, X, k] += r.w * c[i, Y, j] * c[j, Z, k]
 189        return c

 191    def language(self, depth):
 192        "Enumerate strings generated by this cfg by derivations up to a the given `depth`."
 193        lang = self.R.chart()

 305    def trim(self, bottomup_only=False):
 306        "Return an equivalent grammar with no dead or useless nonterminals or rules."
 307
 308        if self._trim_cache[bottomup_only] is not None:
 309            return self._trim_cache[bottomup_only]
 310
 311        C = set(self.V)
 312        C.update(e.head for e in self.rules if len(e.body) == 0)
 313
 314        incoming = defaultdict(list)
 315        outgoing = defaultdict(list)
 316        for e in self:
 317            incoming[e.head].append(e)
 318            for b in e.body:
 319                outgoing[b].append(e)
 320
 321        agenda = set(C)
 322        while agenda:
 323            x = agenda.pop()
 324            for e in outgoing[x]:
 325                if all((b in C) for b in e.body):
 326                    if e.head not in C:
 327                        C.add(e.head)
 328                        agenda.add(e.head)
 329
 330        if bottomup_only:
 331            val = self._trim(C)
 332            self._trim_cache[bottomup_only] = val
 333            val._trim_cache[bottomup_only] = val
 334            return val
 335
 336        T = {self.S}
 337        agenda.update(T)
 338        while agenda:
 339            x = agenda.pop()
 340            for e in incoming[x]:
 341                # assert e.head in T
 342                for b in e.body:
 343                    if b not in T and b in C:
 344                        T.add(b)
 345                        agenda.add(b)
 346
 347        val = self._trim(T)
 348        self._trim_cache[bottomup_only] = val
 349        val._trim_cache[bottomup_only] = val
 350        return val

 367        "Enumerate derivations of symbol X with height <= H"
 368        if X is None:

 392        # compute the matrix closure of the unary rules, so we can unfold them
 393        # into the preterminal and binary rules.
 394        A = WeightedGraph(self.R)
 395        for r in self:

 501    def _push_null_weights(self, null_weight, rename=NotNull):
 502        """
 503        Returns a grammar that generates the same weighted language but it is
 504        nullary-free at all nonterminals except its start symbol.  [Assumes that
 505        S does not appear on any RHS; call separate_start to ensure this.]
 506
 507        The nonterminals with nonzero null_weight will be eliminated from the
 508        grammar.  They will be repaced with nullary-free variants that are
 509        marked according to `rename` (the default option is to mark them with a
 510        dollar sign prefix).
 511
 512        Bonus (Hygiene property): Any nonterminal that survives the this
 513        transformation is guaranteed to generate the same weighted language.
 514        """
 515
 516        # Warning: this method might have issues when `separate_start` hasn't
 517        # been run before.  So we run it rather than leaving it up to chance.
 518        assert self.S not in {y for r in self for y in r.body}
 519
 520        def f(x):
 521            "Rename nonterminal if necessary"
 522            if (
 523                null_weight[x] == self.R.zero or x == self.S
 524            ):  # not necessary; keep old name
 525                return x
 526            else:
 527                return rename(x)
 528
 529        rcfg = self.spawn()
 530        rcfg.add(null_weight[self.S], self.S)
 531
 532        for r in self:
 533            if len(r.body) == 0:
 534                continue  # drop nullary rule
 535
 536            for B in product([0, 1], repeat=len(r.body)):
 537                v, new_body = r.w, []
 538
 539                for i, b in enumerate(B):
 540                    if b:
 541                        v *= null_weight[r.body[i]]
 542                    else:
 543                        new_body.append(f(r.body[i]))
 544
 545                # exclude the cases that would be new nullary rules!
 546                if len(new_body) > 0:
 547                    rcfg.add(v, f(r.head), *new_body)
 548
 549        return rcfg

 551    def separate_start(self):
 552        "Ensure that the start symbol does not appear on the RHS of any rule."
 553        # create a new start symbol if the current one appears on the rhs of any existing rule
 554        if self.S in {y for r in self for y in r.body}:
 555            S = _gen_nt(self.S)
 556            new = self.spawn(S=S)
 557            # preterminal rules
 558            new.add(self.R.one, S, self.S)
 559            for r in self:
 560                new.add(r.w, r.head, *r.body)
 561            return new
 562        else:
 563            return self

 565    def separate_terminals(self):
 566        "Ensure that the each terminal is produced by a preterminal rule."
 567        one = self.R.one
 568        new = self.spawn()
 569
 570        _preterminal = {}
 571
 572        def preterminal(x):
 573            y = _preterminal.get(x)
 574            if y is None:
 575                y = new.add(one, _gen_nt(), x)
 576                _preterminal[x] = y
 577            return y

 579        for r in self:
 580            if len(r.body) == 1 and self.is_terminal(r.body[0]):
 581                new.add(r.w, r.head, *r.body)

 583                new.add(
 584                    r.w,
 585                    r.head,
 586                    *(

 594    def binarize(self):
 595        "Return an equivalent grammar with arity  2."
 596        new = self.spawn()
 597
 598        stack = list(self)
 599        while stack:
 600            p = stack.pop()
 601            if len(p.body) <= 2:
 602                new.add(p.w, p.head, *p.body)
 603            else:
 604                stack.extend(self._fold(p, [(0, 1)]))
 605
 606        return new

 608    def _fold(self, p, I):
 609        # new productions
 610        P, heads = [], []
 611        for i, j in I:
 612            head = _gen_nt()
 613            heads.append(head)
 614            body = p.body[i : j + 1]
 615            P.append(Rule(self.R.one, head, body))
 616
 617        # new "head" production
 618        body = tuple()
 619        start = 0
 620        for (end, n), head in zip(I, heads):
 621            body += p.body[start:end] + (head,)
 622            start = n + 1
 623        body += p.body[start:]
 624        P.append(Rule(p.w, p.head, body))
 625
 626        return P

 854    def _compose_bottom_up_epsilon(self, fst):
 855        "Determine which items of the composition grammar are supported"
 856
 857        A = set()
 858
 859        I = defaultdict(set)  # incomplete items
 860        C = defaultdict(set)  # complete items
 861        R = defaultdict(set)  # rules indexed by first subgoal; non-nullary
 862
 863        special_rules = [Rule(self.R.one, a, (EPSILON, a)) for a in self.V] + [
 864            Rule(self.R.one, Other(self.S), (self.S,)),
 865            Rule(self.R.one, Other(self.S), (Other(self.S), EPSILON)),
 866        ]
 867
 868        for r in itertools.chain(self, special_rules):
 869            if len(r.body) > 0:
 870                R[r.body[0]].add(r)
 871
 872        # we have two base cases:
 873        #
 874        # base case 1: arcs
 875        for i, (a, _), j, _ in fst.arcs():
 876            A.add((i, a, (), j))  # empty tuple -> the rule 'complete'
 877
 878        # base case 2: nullary rules
 879        for r in self:
 880            if len(r.body) == 0:
 881                for i in fst.states:
 882                    A.add((i, r.head, (), i))
 883
 884        # drain the agenda
 885        while A:
 886            (i, X, Ys, j) = A.pop()
 887
 888            # No pending items ==> the item is complete
 889            if not Ys:
 890                if j in C[i, X]:
 891                    continue
 892                C[i, X].add(j)
 893
 894                # combine the newly completed item with incomplete rules that are
 895                # looking for an item like this one
 896                for h, X1, Zs in I[i, X]:
 897                    A.add((h, X1, Zs[1:], j))
 898
 899                # initialize rules that can start with an item like this one
 900                for r in R[X]:
 901                    A.add((i, r.head, r.body[1:], j))
 902
 903            # Still have pending items ==> advanced the pending items
 904            else:
 905                if (i, X, Ys) in I[j, Ys[0]]:
 906                    continue
 907                I[j, Ys[0]].add((i, X, Ys))

307        "Return an equivalent grammar with no dead or useless nonterminals or rules."
308
309        if self._trim_cache[bottomup_only] is not None:

312        C = set(self.V)
313        C.update(e.head for e in self.rules if len(e.body) == 0)
314
315        incoming = defaultdict(list)
316        outgoing = defaultdict(list)
317        for e in self:
318            incoming[e.head].append(e)
319            for b in e.body:
320                outgoing[b].append(e)
321
322        agenda = set(C)

502    def _push_null_weights(self, null_weight, rename=NotNull):
503        """
504        Returns a grammar that generates the same weighted language but it is
505        nullary-free at all nonterminals except its start symbol.  [Assumes that
506        S does not appear on any RHS; call separate_start to ensure this.]
507
508        The nonterminals with nonzero null_weight will be eliminated from the
509        grammar.  They will be repaced with nullary-free variants that are
510        marked according to `rename` (the default option is to mark them with a
511        dollar sign prefix).
512
513        Bonus (Hygiene property): Any nonterminal that survives the this
514        transformation is guaranteed to generate the same weighted language.
515        """
516
517        # Warning: this method might have issues when `separate_start` hasn't
518        # been run before.  So we run it rather than leaving it up to chance.
519        assert self.S not in {y for r in self for y in r.body}
520
521        def f(x):
522            "Rename nonterminal if necessary"
523            if (
524                null_weight[x] == self.R.zero or x == self.S
525            ):  # not necessary; keep old name
526                return x
527            else:
528                return rename(x)
529
530        rcfg = self.spawn()
531        rcfg.add(null_weight[self.S], self.S)
532
533        for r in self:
534            if len(r.body) == 0:
535                continue  # drop nullary rule
536
537            for B in product([0, 1], repeat=len(r.body)):
538                v, new_body = r.w, []
539
540                for i, b in enumerate(B):
541                    if b:
542                        v *= null_weight[r.body[i]]
543                    else:
544                        new_body.append(f(r.body[i]))
545
546                # exclude the cases that would be new nullary rules!
547                if len(new_body) > 0:
548                    rcfg.add(v, f(r.head), *new_body)
549
550        return rcfg

596        "Return an equivalent grammar with arity  2."

609    def _fold(self, p, I):
610        # new productions
611        P, heads = [], []
612        for i, j in I:
613            head = _gen_nt()
614            heads.append(head)
615            body = p.body[i : j + 1]
616            P.append(Rule(self.R.one, head, body))
617
618        # new "head" production
619        body = tuple()
620        start = 0
621        for (end, n), head in zip(I, heads):
622            body += p.body[start:end] + (head,)
623            start = n + 1
624        body += p.body[start:]
625        P.append(Rule(p.w, p.head, body))
626
627        return P

714    # def agenda(self, tol=1e-12, maxiter=float('inf')):
715    def agenda(self, tol=1e-12, maxiter=100_000):
716        "Agenda-based semi-naive evaluation for treesums."
717        old = self.R.chart()
718
719        # precompute the mapping from updates to where they need to go
720        routing = defaultdict(list)
721        for r in self:
722            for k in range(len(r.body)):
723                routing[r.body[k]].append((r, k))
724
725        deps = self.dependency_graph()
726        blocks = deps.blocks
727        bucket = deps.buckets
728
729        # helper function
730        def update(x, W):
731            change[bucket[x]][x] += W
732
733        change = defaultdict(self.R.chart)
734        for a in self.V:
735            update(a, self.R.one)
736
737        for r in self:
738            if len(r.body) == 0:
739                update(r.head, r.w)
740
741        b = len(blocks)
742        iteration = 0
743        while b >= 0:
744            iteration += 1
745
746            # Move on to the next block
747            if len(change[b]) == 0 or iteration > maxiter:
748                b -= 1
749                iteration = 0  # reset iteration number for the next bucket
750                continue
751
752            u, v = change[b].popitem()
753
754            new = old[u] + v
755
756            if self.R.metric(old[u], new) <= tol:
757                continue
758
759            for r, k in routing[u]:
760                W = r.w
761                for j in range(len(r.body)):
762                    if u == r.body[j]:
763                        if j < k:
764                            W *= new
765                        elif j == k:
766                            W *= v
767                        else:
768                            W *= old[u]
769                    else:
770                        W *= old[r.body[j]]
771
772                update(r.head, W)
773
774            old[u] = new
775
776        return old

778    def naive_bottom_up(self, *, tol=1e-12, timeout=100_000):
779        "Naive bottom-up evaluation for treesums; better to use `agenda`."
780
781        def _approx_equal(U, V):
782            return all((self.R.metric(U[X], V[X]) <= tol) for X in self.N)
783
784        R = self.R
785        V = R.chart()
786        counter = 0
787        while counter < timeout:
788            U = self._bottom_up_step(V)
789            if _approx_equal(U, V):
790                break
791            V = U
792            counter += 1
793        return V

796        R = self.R

798        U = R.chart()
799        for a in self.V:

804                if self.is_nonterminal(X):

855    def _compose_bottom_up_epsilon(self, fst):
856        "Determine which items of the composition grammar are supported"
857
858        A = set()
859
860        I = defaultdict(set)  # incomplete items
861        C = defaultdict(set)  # complete items
862        R = defaultdict(set)  # rules indexed by first subgoal; non-nullary
863
864        special_rules = [Rule(self.R.one, a, (EPSILON, a)) for a in self.V] + [
865            Rule(self.R.one, Other(self.S), (self.S,)),
866            Rule(self.R.one, Other(self.S), (Other(self.S), EPSILON)),
867        ]
868
869        for r in itertools.chain(self, special_rules):
870            if len(r.body) > 0:
871                R[r.body[0]].add(r)
872
873        # we have two base cases:
874        #
875        # base case 1: arcs
876        for i, (a, _), j, _ in fst.arcs():
877            A.add((i, a, (), j))  # empty tuple -> the rule 'complete'
878
879        # base case 2: nullary rules
880        for r in self:
881            if len(r.body) == 0:
882                for i in fst.states:
883                    A.add((i, r.head, (), i))
884
885        # drain the agenda
886        while A:
887            (i, X, Ys, j) = A.pop()
888
889            # No pending items ==> the item is complete
890            if not Ys:
891                if j in C[i, X]:
892                    continue
893                C[i, X].add(j)
894
895                # combine the newly completed item with incomplete rules that are
896                # looking for an item like this one
897                for h, X1, Zs in I[i, X]:
898                    A.add((h, X1, Zs[1:], j))
899
900                # initialize rules that can start with an item like this one
901                for r in R[X]:
902                    A.add((i, r.head, r.body[1:], j))
903
904            # Still have pending items ==> advanced the pending items
905            else:
906                if (i, X, Ys) in I[j, Ys[0]]:
907                    continue
908                I[j, Ys[0]].add((i, X, Ys))
909
910                for k in C[j, Ys[0]]:
911                    A.add((i, X, Ys[1:], k))
912
913        return C

915    def __matmul__(self, fst):
916        "Return a CFG denoting the pointwise product or composition of `self` and `fs`."
917
918        # coerce something sequence like into a diagonal FST
919        if isinstance(fst, (str, tuple)):
920            fst = FST.from_string(fst, self.R)
921        # coerce something FSA-like into an FST, might throw an error
922        if not isinstance(fst, FST):
923            fst = fst.to_fst()
924
925        # Initialize the new CFG:
926        # - its start symbol is chosen arbitrarily to be `self.S`
927        # - its the alphabet changes - it is now 'output' alphabet of the transducer
928        new_start = self.S
929        new = self.spawn(S=new_start, V=fst.B - {EPSILON})
930
931        # The bottom-up intersection algorithm is a two-pass algorithm
932        #
933        # Pass 1: Determine the set of items that are possiblly nonzero-valued
934        C = self._compose_bottom_up_epsilon(fst)
935
936        special_rules = [Rule(self.R.one, a, (EPSILON, a)) for a in self.V] + [
937            Rule(self.R.one, Other(self.S), (self.S,)),
938            Rule(self.R.one, Other(self.S), (Other(self.S), EPSILON)),
939        ]
940
941        def join(start, Ys):
942            """
943            Helper method; expands the rule body
944
945            Given Ys = [Y_1, ... Y_K], we will enumerate expansion of the form
946
947            (s_0, Y_1, s_1), (s_1, Y_2, s_2), ..., (s_{k-1}, Y_K, s_K)
